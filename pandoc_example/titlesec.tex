% Fix the appearance of figures
\usepackage{float}
\floatplacement{figure}{H}

\usepackage[utf8]{inputenc}
\usepackage{pifont}
\usepackage{newunicodechar}

\usepackage{xcolor}
\definecolor{links}{HTML}{2A1B81}
\hypersetup{colorlinks,linkcolor=,urlcolor=links}

    keepspaces=true, 
    showspaces=false, 
    showstringspaces=false, 
    breaklines=true, 
    frame=tb
